/**
  ******************************************************************************
  * @file    main.c
  * @brief   Lightning Rod Structural Health Monitoring System - 主控调度引擎
  * @date    2026-03-08
  * @note    三层速率架构：
  * 1. 硬件层 (6400Hz后台): TIM2 + DMA 自动搬运 8 路波形，不占 CPU。
  * 2. 业务层 (50Hz前台): Process_Cycle_Task 提取 128 点波形算有效值。
  * 3. 轮询层 (1Hz/空闲): W5500 以太网分发、PC5 温漂补偿、LED 心跳。
  ******************************************************************************
  */

// --- 系统底层驱动库 ---
#include "stm32f4xx.h"
#include "sys.h"
#include "sys_tick.h"      // 1ms 系统心跳基准
#include "usart.h"         // 串口驱动
#include "led.h"           // LED 指示灯

// --- 核心业务与通讯模块 ---
#include "sys_config.h"          // 全局参数与 512 寄存器池
#include "current_monitoring.h"  // 高频交流电流采样 (PA0-PA7)
#include "sensor.h"              // 慢速模拟量采样 (PC5 温漂/环境)
#include "w5500_driver.h"        // W5500 以太网硬件层
#include "protocol_dispatcher.h" // 协议分发 (Modbus TCP/RTU & 0xAA 全量包)
#include "modbus_slave.h"        // Modbus 解析引擎 

int main(void)
{ 
    /* ================================================================== */
    /* [一] 局部时间标记初始化 (非阻塞调度核心)                           */
    /* ================================================================== */
    uint32_t last_blink_time = 0;     // 用于 LED 500ms 翻转
    uint32_t last_sensor_time = 0;    // 用于 1Hz 慢速采样

    /* ================================================================== */
    /* [二] 硬件与基础服务初始化序列                                      */
    /* ================================================================== */
    SysTick_Timer_Init();    // 1. 开启 1ms 系统心跳
    LED_Init();              // 2. 运行指示灯初始化
    uart_init(115200);       // 3. 调试/RTU 串口初始化
    
    // 4. 业务配置初始化 (从 Flash 读取出厂参数、变比、IP 等，填入寄存器)
    SYS_Init();              
    
    // 5. 慢速传感器硬件初始化 (配置 PC5 引脚与独立的低速 ADC 规则) 
    Sensor_Init();           

    // 6. 以太网硬件层初始化 (配置 GPIOD 引脚，复位 W5500)
    W5500_GPIO_Init();       

    /* ================================================================== */
    /* [三] 核心引擎启动 (启动后进入自动运行状态)                         */
    /* ================================================================== */
    // 1. 网络引擎启动
    W5500_Sync_Params();     // 同步 11-22 号寄存器里的 IP/Port 到硬件芯片
    W5500_Socket_Open();     // 打开 Socket 0，监听 502 端口
    
    // 2. 采集引擎启动 (放在最后，防止中断打断前面的网络配置)
    ADC_DMA_Config();        // 启动 TIM2(6400Hz) 和 DMA 乒乓搬运

    /* ================================================================== */
    /* [四] 超级主循环 (非阻塞式任务轮询)                                 */
    /* ================================================================== */
    while(1)
    {
        // 获取当前系统毫秒数，供各个任务判断时间片
        uint32_t current_time = Get_System_Time(); 

        /* -------------------------------------------------------------
         * 任务 A: 极速波形加工任务 (50Hz 节拍)
         * 内部带有 latest_ready_idx 锁。平时瞬间退出，DMA凑齐128点时执行。
         *
         * ------------------------------------------------------------- */
        Process_Cycle_Task(); 

        /* -------------------------------------------------------------
         * 任务 B: 网络与通讯分发任务 (全速轮询)
         * 负责处理以太网拉取请求，生成包含 138 个寄存器与波形的 2500 字节大包。
         *
         * ------------------------------------------------------------- */
        W5500_Process(); 
        
        /* -------------------------------------------------------------
         * 任务 C: 配置修改与存取监控 (全速轮询)
         * 监控 99/100 号寄存器，若被上位机写 1，则触发 Flash 存取。
         *
         * ------------------------------------------------------------- */
        SYS_Monitor_Cmd();

        /* -------------------------------------------------------------
         * 任务 D: 慢速环境传感器监控 (1Hz 节拍)
         * 处理晶振附近的 PC5 温度采样及温漂系数运算。
         * ------------------------------------------------------------- */
        if(current_time - last_sensor_time >= 1000) 
        {
            last_sensor_time = current_time; 
            Sensor_Update_Task(); 
        }

        /* -------------------------------------------------------------
         * 任务 E: LED 系统运行指示灯 (1Hz 周期，500ms 翻转)
         *
         * ------------------------------------------------------------- */
        if(current_time - last_blink_time >= 1000) 
        {
            last_blink_time = current_time;    
        }
        
        // 前 500ms 亮(0)，后 500ms 灭(1)
        if(current_time - last_blink_time < 500) {
            LED = 0; 
        } else {
            LED = 1; 
        }
    }
}
